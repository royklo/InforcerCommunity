    ```javascript
    function highlightBash(code) {
        var text = code.textContent;
        var tokens = [];
        var re = /(#[^\n]*|"(?:[^"\\]|\\.)*"|'[^']*'|\$\{[\w]+\}|\$[\w]+|\b(?:if|then|else|elif|fi|for|do|done|while|until|case|esac|function|return|local|export|set|trap|in)\b|\b(?:echo|curl|rm|mkdir|cp|mv|chmod|chown|grep|sed|awk|cat|ls|cd|pwd|source|eval|exec|exit)\b)/g;
        var lastIdx = 0, m;
        while ((m = re.exec(text)) !== null) {
            if (m.index > lastIdx) tokens.push(escHtml(text.substring(lastIdx, m.index)));
            var t = m[0], cls = '';
            if (t[0] === '#') cls = 'sh-comment';
            else if (t[0] === '"' || t[0] === "'") cls = 'sh-string';
            else if (t[0] === '$') cls = 'sh-variable';
            else if (/^(if|then|else|elif|fi|for|do|done|while|until|case|esac|function|return|local|export|set|trap|in)$/.test(t)) cls = 'sh-keyword';
            else cls = 'sh-command';
            tokens.push('<span class="' + cls + '">' + escHtml(t) + '</span>');
            lastIdx = m.index + t.length;
        }
        if (lastIdx < text.length) tokens.push(escHtml(text.substring(lastIdx)));
        code.innerHTML = tokens.join('');
    }
    ```
