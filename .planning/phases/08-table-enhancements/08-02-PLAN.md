    ```javascript
    // Column resize (Phase 8 TBL-01, per D-01 through D-04)
    try {
    (function() {
      var table = document.getElementById('comparison-table');
      if (!table) return;
      var ths = table.querySelectorAll('thead th');
      var defaultWidths = [];
      var minWidths = [];

      // Capture default widths after layout (per D-03)
      ths.forEach(function(th, i) {
        defaultWidths[i] = th.offsetWidth;
        minWidths[i] = (i === 0) ? 40 : 60;
      });

      // Set table-layout: fixed after capturing widths (Pitfall 1)
      ths.forEach(function(th) { th.style.width = th.offsetWidth + 'px'; });
      table.style.tableLayout = 'fixed';

      ths.forEach(function(th, i) {
        var handle = document.createElement('div');
        handle.className = 'col-resize-handle';
        th.appendChild(handle);

        // Double-click resets to default width (per D-03)
        handle.addEventListener('dblclick', function(e) {
          e.stopPropagation();
          e.preventDefault();
          th.style.width = defaultWidths[i] + 'px';
        });

        var startX, startW;
        handle.addEventListener('mousedown', function(e) {
          e.stopPropagation();
          e.preventDefault();
          startX = e.pageX;
          startW = th.offsetWidth;
          handle.classList.add('resizing');
          document.body.style.cursor = 'col-resize';
          document.addEventListener('mousemove', onMouseMove);
          document.addEventListener('mouseup', onMouseUp);
        });

        function onMouseMove(e) {
          var w = Math.max(minWidths[i], startW + (e.pageX - startX));
          th.style.width = w + 'px';
        }
        function onMouseUp() {
          handle.classList.remove('resizing');
          document.body.style.cursor = '';
          document.removeEventListener('mousemove', onMouseMove);
          document.removeEventListener('mouseup', onMouseUp);
        }
      });
    })();
    } catch(e) { console.warn('Column resize error:', e); }
    ```

    This JS is emitted via `[void]$sb.AppendLine(...)` calls, one line per AppendLine, matching the existing pattern used for highlightPS, highlightBash, and applyFilters JS functions.

    IMPORTANT: The `e.stopPropagation()` in mousedown prevents the `th[onclick]` sort handler from firing during resize (Pitfall 2 from RESEARCH.md).
