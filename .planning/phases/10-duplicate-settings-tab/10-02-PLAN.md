**JS Addition 2: analyzeDuplicate() function (per D-07, D-09 — direct port from DuplicateSettingsTab.tsx lines 17-57)**

Port the TypeScript function to ES5-compatible vanilla JS. CRITICAL: Use PascalCase property access (`p.Policy`, `p.Value`, `p.Side`) because PowerShell's ConvertTo-Json preserves the original PascalCase from the Phase 4 data contract.

```javascript
function analyzeDuplicate(policies) {
    var uniqueValues = new Set(policies.map(function(p) { return p.Value; }));
    var srcEntries = policies.filter(function(p) { return p.Side === 'Source'; });
    var dstEntries = policies.filter(function(p) { return p.Side === 'Destination'; });
    var hasBothSides = srcEntries.length > 0 && dstEntries.length > 0;
    var srcNames = new Set(srcEntries.map(function(p) { return p.Policy.toLowerCase(); }));
    var matchedPairs = dstEntries.filter(function(d) { return srcNames.has(d.Policy.toLowerCase()); });
    var crossTenantMatch = matchedPairs.length > 0 && matchedPairs.some(function(d) {
        var src = srcEntries.find(function(s) { return s.Policy.toLowerCase() === d.Policy.toLowerCase(); });
        return src && src.Value === d.Value;
    });
    var valueCounts = new Map();
    for (var i = 0; i < policies.length; i++) {
        var v = policies[i].Value;
        valueCounts.set(v, (valueCounts.get(v) || 0) + 1);
    }
    var majorityEntry = null;
    valueCounts.forEach(function(count, val) {
        if (!majorityEntry || count > majorityEntry[1]) majorityEntry = [val, count];
    });
    var majorityValue = majorityEntry ? majorityEntry[0] : null;
    var outliers = policies.filter(function(p) { return p.Value !== majorityValue; });
    if (!hasBothSides) {
        if (outliers.length > 0) {
            var outlierNames = outliers.map(function(p) { return p.Policy; }).join(', ');
            return policies.length + ' policies in the same tenant configure this setting with different values. "' + outlierNames + '" differs from the others. If assignments overlap, Intune will report a conflict and the setting may not apply until the conflict is resolved.';
        }
        return policies.length + ' policies configure this setting differently within the same tenant. If assignments overlap, Intune will report a conflict and the setting may not apply until resolved.';
    }
    if (crossTenantMatch && outliers.length > 0) {
        var outlierNames2 = outliers.map(function(p) { return p.Policy + ' (' + p.Side + ')'; }).join(', ');
        return 'The cross-tenant comparison matches, but ' + outlierNames2 + ' has a different value. If its assignments overlap with the matching policies, Intune will report a conflict and the setting may not apply until resolved.';
    }
    if (uniqueValues.size === policies.length) {
        return 'Every policy has a unique value for this setting. Review which value should be the standard across both tenants.';
    }
    return 'This setting is configured in ' + policies.length + ' policies across both tenants with ' + uniqueValues.size + ' different values. Review to ensure consistency.';
}

**JS Addition 3: dupTabSearch() function (per D-10, D-11, D-12)**

function dupTabSearch(query) {
    var q = query.toLowerCase();
    var rows = document.querySelectorAll('#dup-table-body tr');
    var shown = 0;
    rows.forEach(function(row) {
        var setting = (row.getAttribute('data-setting') || '').toLowerCase();
        var policies = (row.getAttribute('data-policies') || '').toLowerCase();
        var match = !q || setting.indexOf(q) >= 0 || policies.indexOf(q) >= 0;
        row.style.display = match ? '' : 'none';
        if (match) shown++;
    });
    var summary = document.getElementById('dup-summary');
    if (summary) {
        summary.innerHTML = 'Showing <strong>' + shown + '</strong> of <strong>' + rows.length + '</strong> duplicate settings across <strong>' + dupPolicyCount + '</strong> policies';
    }
    var noResults = document.getElementById('dup-no-results');
    if (noResults) noResults.style.display = (shown === 0 && q) ? '' : 'none';
}
```
